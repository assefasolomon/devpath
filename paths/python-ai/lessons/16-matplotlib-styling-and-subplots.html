A chart that communicates clearly is more valuable than one that is merely correct. Professional data scientists spend time making their visualisations clean, consistent, and readable. This lesson teaches you how to use Matplotlib's styling system, create multi-panel figures with subplots, and customise every element of a chart so your work looks publication-ready.

Lesson Overview
  • Apply built-in styles with plt.style.use().
  • Create multi-panel figures with plt.subplots(rows, cols).
  • Share axes between subplots.
  • Customise tick marks, tick labels, and spines.
  • Add annotations and text to charts.
  • Use plt.tight_layout() and plt.subplots_adjust() for spacing.

Built-in Styles

Matplotlib ships with dozens of built-in styles. Applying a style changes the background, colours, fonts, and grid in one line — no manual configuration needed.

import matplotlib.pyplot as plt
import numpy as np

# List all available styles
print(plt.style.available)

# Apply a style — must be called before creating the figure
plt.style.use("seaborn-v0_8-darkgrid")

x = np.linspace(0, 10, 100)
fig, ax = plt.subplots(figsize=(8, 4))
ax.plot(x, np.sin(x), label="sin(x)")
ax.plot(x, np.cos(x), label="cos(x)")
ax.set_title("Styled Chart — seaborn-darkgrid")
ax.legend()
plt.tight_layout()
plt.show()

# Other useful styles:
# "ggplot"           — R's ggplot2 look
# "fivethirtyeight"  — FiveThirtyEight journalism style
# "bmh"              — Bayesian Methods for Hackers
# "seaborn-v0_8-whitegrid"  — clean white background
# "dark_background"  — dark theme
💡 Use with plt.style.context("ggplot"): to apply a style only inside a block without changing the global default for the rest of your script.

Creating Subplots

plt.subplots(rows, cols) creates a grid of axes. This is how you build a dashboard of related charts in a single figure.

import matplotlib.pyplot as plt
import numpy as np

np.random.seed(42)
x = np.linspace(0, 10, 100)

# 2 rows, 2 columns — 4 subplots
fig, axes = plt.subplots(2, 2, figsize=(10, 7))

# axes is a 2D array — access with [row, col]
axes[0, 0].plot(x, np.sin(x), color="steelblue")
axes[0, 0].set_title("Line: sin(x)")

axes[0, 1].bar(["A", "B", "C", "D"],
               [0.85, 0.92, 0.78, 0.95],
               color="tomato")
axes[0, 1].set_title("Bar Chart")

axes[1, 0].scatter(np.random.randn(50),
                   np.random.randn(50),
                   alpha=0.6, color="mediumseagreen")
axes[1, 0].set_title("Scatter Plot")

axes[1, 1].hist(np.random.normal(0, 1, 300),
                bins=20, color="mediumpurple", edgecolor="white")
axes[1, 1].set_title("Histogram")

# Add a title for the whole figure
fig.suptitle("Data Visualisation Dashboard", fontsize=16, fontweight="bold")

plt.tight_layout()
plt.savefig("dashboard.png", dpi=150, bbox_inches="tight")
plt.show()

Shared Axes

When subplots show related data, sharing axes makes comparison easier and keeps scales consistent.

import matplotlib.pyplot as plt
import numpy as np

epochs = np.arange(1, 31)
np.random.seed(0)
train_loss = 2.0 * np.exp(-0.1 * epochs) + np.random.normal(0, 0.05, 30)
val_loss   = 2.0 * np.exp(-0.09 * epochs) + np.random.normal(0, 0.07, 30)
train_acc  = 1 - np.exp(-0.12 * epochs) + np.random.normal(0, 0.02, 30)
val_acc    = 1 - np.exp(-0.10 * epochs) + np.random.normal(0, 0.03, 30)

# Share x-axis — both plots scroll together
fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(9, 6), sharex=True)

ax1.plot(epochs, train_loss, label="Train Loss", color="steelblue")
ax1.plot(epochs, val_loss, label="Val Loss", color="tomato", linestyle="--")
ax1.set_ylabel("Loss")
ax1.legend()
ax1.grid(True, alpha=0.3)
ax1.set_title("Training Curves")

ax2.plot(epochs, train_acc, label="Train Acc", color="steelblue")
ax2.plot(epochs, val_acc, label="Val Acc", color="tomato", linestyle="--")
ax2.set_ylabel("Accuracy")
ax2.set_xlabel("Epoch")
ax2.legend()
ax2.grid(True, alpha=0.3)

plt.tight_layout()
plt.savefig("training_curves.png", dpi=150, bbox_inches="tight")
plt.show()

Customising Ticks and Spines

import matplotlib.pyplot as plt
import numpy as np

fig, ax = plt.subplots(figsize=(8, 4))
x = np.linspace(0, 2 * np.pi, 100)
ax.plot(x, np.sin(x), color="steelblue", linewidth=2)

# Custom tick positions and labels
ax.set_xticks([0, np.pi/2, np.pi, 3*np.pi/2, 2*np.pi])
ax.set_xticklabels(["0", "π/2", "π", "3π/2", "2π"], fontsize=11)

# Tick direction and size
ax.tick_params(axis="both", direction="in", length=6, width=1.2)

# Remove top and right spines for a cleaner look
ax.spines["top"].set_visible(False)
ax.spines["right"].set_visible(False)

ax.set_title("Sine Wave — Custom Ticks")
ax.set_ylabel("sin(x)")
plt.tight_layout()
plt.show()

Annotations and Text

Annotations draw the reader's attention to a specific point or region in the chart. Use them to highlight maximums, thresholds, or key events.

import matplotlib.pyplot as plt
import numpy as np

np.random.seed(1)
epochs = np.arange(1, 21)
loss = 2.0 * np.exp(-0.2 * epochs) + np.random.normal(0, 0.04, 20)
best_epoch = np.argmin(loss) + 1
best_loss  = loss[best_epoch - 1]

fig, ax = plt.subplots(figsize=(9, 4))
ax.plot(epochs, loss, color="steelblue", linewidth=2)

# Annotate the best point with an arrow
ax.annotate(
    f"Best: {best_loss:.3f}",
    xy=(best_epoch, best_loss),
    xytext=(best_epoch + 3, best_loss + 0.3),
    arrowprops=dict(arrowstyle="->", color="tomato"),
    fontsize=11, color="tomato"
)

# Mark the point with a dot
ax.scatter([best_epoch], [best_loss], color="tomato", zorder=5, s=80)

# Add a horizontal threshold line
ax.axhline(0.3, color="gray", linestyle=":", linewidth=1.5,
           label="Target threshold")

ax.set_title("Loss with Best Epoch Annotated")
ax.set_xlabel("Epoch")
ax.set_ylabel("Loss")
ax.legend()
ax.grid(True, alpha=0.3)
plt.tight_layout()
plt.savefig("annotated_loss.png", dpi=150, bbox_inches="tight")
plt.show()

Layout and Spacing

import matplotlib.pyplot as plt
import numpy as np

fig, axes = plt.subplots(1, 3, figsize=(12, 4))
x = np.linspace(0, 10, 100)

for i, ax in enumerate(axes):
    ax.plot(x, np.sin(x + i), color=["steelblue", "tomato", "seagreen"][i])
    ax.set_title(f"Panel {i+1}")
    ax.set_xlabel("x")
    ax.grid(True, alpha=0.3)

# tight_layout — auto-fix overlapping labels
plt.tight_layout()

# Or fine-tune manually
# plt.subplots_adjust(left=0.05, right=0.95,
#                     top=0.90, bottom=0.12,
#                     wspace=0.3, hspace=0.4)

fig.suptitle("Three Panel Figure", fontsize=14, y=1.02)
plt.savefig("three_panels.png", dpi=150, bbox_inches="tight")
plt.show()
